# Автоматическая проверка и инициализация создания БД:
✅ Как сделать автоматически (правильный способ):
В storage/database.py добавить функцию ensure_db():

def ensure_db():
    """Проверяет наличие БД и создаёт её при необходимости"""
    import os
    from config import config
    
    if not os.path.exists(config.DATABASE_PATH):
        print(f"📁 База данных не найдена. Создаю {config.DATABASE_PATH}")
        init_db()
        seed_data()  # только для разработки
        print("✅ База данных создана")
    else:
        print("✅ База данных уже существует")


    В bot.py при запуске:

    from storage.database import ensure_db

def main():
    ensure_db()  # Проверка при каждом запуске
    # ... остальной код

🎯 Как работают слоты в реальном бизнесе
В реальном проекте слоты не добавляют вручную. Они управляются динамически через админ-панель или интеграцию с календарём.

📋 Три основных подхода в реальных проектах
1. Админ-панель (самый частый)
Администратор через интерфейс (веб или чат-бот) управляет расписанием:

Действие	Как реализовано
Добавить слот	Админ выбирает дату и время → бот создаёт запись в БД
Удалить слот	Админ видит сетку расписания → удаляет ненужные слоты
Скопировать расписание	На следующую неделю копируются слоты с текущей
Массовое создание	Указать интервал: "с 10:00 до 18:00, шаг 1 час, на 7 дней"
Технически: веб-интерфейс на Django/Flask или отдельный чат-бот для администратора с кнопками.

2. Интеграция с Google Calendar / Яндекс.Расписания
Слоты синхронизируются с внешним календарём:

Плюсы	Минусы
Бесплатно	Требует настройки API
Мастера уже пользуются календарём	Ограничения на количество запросов
Не нужно писать админ-панель	Требуется OAuth авторизация
Как работает: Бот смотрит свободное время в Google Calendar мастера и показывает его пользователю. При записи создаёт событие в календаре.

3. Гибридный подход (рекомендую для портфолио)
Слоты генерируются автоматически на основе шаблона, но управляются через бота-администратора:

# Пример: генерация слотов на неделю вперёд
def generate_weekly_slots(master_id, start_date, work_hours, step_minutes=60):
    """
    start_date: '2026-06-10'
    work_hours: [(10, 18)]  # с 10 до 18
    step_minutes: 60  # каждый час
    """
    slots = []
    for day_offset in range(7):
        date = start_date + timedelta(days=day_offset)
        if date.weekday() in [5, 6]:  # выходные
            continue
        for hour in range(work_hours[0][0], work_hours[0][1]):
            start_time = date.replace(hour=hour, minute=0)
            end_time = start_time + timedelta(minutes=step_minutes)
            slots.append((master_id, start_time, end_time))
    return slots
